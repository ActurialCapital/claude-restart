4. Replace the process-kill section (the PPID walk) with instance-aware logic. After writing the restart file and printing confirmation:
```bash
# Kill/restart the target instance
if [[ -n "$INSTANCE" ]]; then
    # Per D-13: use systemctl for instance restart (no PID hunting)
    systemctl --user restart "claude@${INSTANCE}.service" 2>/dev/null || {
        echo "claude-restart: warning: systemctl restart failed for claude@${INSTANCE}" >&2
        echo "claude-restart: restart file written; restart will occur when instance exits" >&2
    }
elif [[ -n "${CLAUDE_RESTART_TARGET_PID:-}" ]]; then
    # Testing override: use specified PID directly
    target_pid="$CLAUDE_RESTART_TARGET_PID"
    kill -TERM "$target_pid" 2>/dev/null || true
else
    # Fallback: PPID walk for non-systemd environments (macOS dev)
    target_pid=""
    current_pid="$$"
    for i in $(seq 1 5); do
        parent=$(ps -o ppid= -p "$current_pid" 2>/dev/null | tr -d ' ')
        if [[ -z "$parent" || "$parent" == "0" || "$parent" == "1" ]]; then
            break
        fi
        parent_cmd=$(ps -o command= -p "$parent" 2>/dev/null || true)
        if [[ "$parent_cmd" == *"node"* && "$parent_cmd" == *"claude"* ]]; then
            target_pid="$parent"
            break
        fi
        current_pid="$parent"
    done
    if [[ -n "$target_pid" ]]; then
        kill -TERM "$target_pid" 2>/dev/null || true
    else
        echo "claude-restart: warning: could not find claude process to kill" >&2
        echo "claude-restart: restart file written; restart will occur when claude exits" >&2
    fi
fi
```
