Create `bin/claude-restart` as a bash script:

```
#!/bin/bash
# claude-restart: trigger a claude restart with new CLI options
set -euo pipefail

RESTART_FILE="${CLAUDE_RESTART_FILE:-$HOME/.claude-restart}"
DEFAULT_OPTS="${CLAUDE_RESTART_DEFAULT_OPTS:-}"

# Determine options to write
if [[ $# -gt 0 ]]; then
    # Args provided: use them
    opts="$*"
else
    # No args: use default opts (may be empty)
    opts="$DEFAULT_OPTS"
fi

# Write restart file
# Non-empty opts = file with content, empty = empty file (wrapper uses original args)
if [[ -n "$opts" ]]; then
    echo "$opts" > "$RESTART_FILE"
else
    touch "$RESTART_FILE"
fi

# Print confirmation
if [[ -n "$opts" ]]; then
    echo "Restarting claude with: $opts"
else
    echo "Restarting claude..."
fi

# Find and kill claude process
# If CLAUDE_RESTART_TARGET_PID is set, use it directly (for testing)
if [[ -n "${CLAUDE_RESTART_TARGET_PID:-}" ]]; then
    target_pid="$CLAUDE_RESTART_TARGET_PID"
else
    # Walk PPID chain to find claude (node) process
    # Process chain: wrapper -> claude (node) -> bash (sandbox) -> this script
    # So we walk: $PPID (bash sandbox) -> its parent (claude/node)
    target_pid=""
    current_pid="$$"

    # Walk up the process tree looking for a node process running claude
    # Maximum 5 levels to prevent infinite loops
    for i in $(seq 1 5); do
        parent=$(ps -o ppid= -p "$current_pid" 2>/dev/null | tr -d ' ')
        if [[ -z "$parent" || "$parent" == "0" || "$parent" == "1" ]]; then
            break
        fi
        # Check if this parent is a node process running claude
        parent_cmd=$(ps -o command= -p "$parent" 2>/dev/null || true)
        if [[ "$parent_cmd" == *"node"* && "$parent_cmd" == *"claude"* ]]; then
            target_pid="$parent"
            break
        fi
        current_pid="$parent"
    done
fi

if [[ -n "$target_pid" ]]; then
    kill -TERM "$target_pid" 2>/dev/null || true
else
    echo "claude-restart: warning: could not find claude process to kill" >&2
    echo "claude-restart: restart file written; restart will occur when claude exits" >&2
fi
```

Make executable with `chmod +x bin/claude-restart`.

Then fix any test failures by adjusting either the script or the tests (minor tweaks only — the test is the spec).

Key implementation details:
- Use `$*` (not `$@`) when writing to file since wrapper reads it as a single line with `cat` then splits with `read -ra`
- PPID walk checks `ps -o command=` for both "node" and "claude" to identify the right process
- Graceful degradation: if PID not found, still write file and print warning (user might be testing outside wrapper)
- `kill -TERM` with `|| true` to handle race conditions where process already exited
