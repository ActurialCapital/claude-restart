**4. Add `do_add` function** (per D-03, D-04, D-05, D-06):
```bash
do_add() {
    local name="$1"
    local git_url="$2"
    local env_dir="$CONFIG_DIR/$name"
    local work_dir="$HOME/instruments/$name"
    local template="$CONFIG_DIR/env.template"

    # Validate name (per Pitfall 3: alphanumeric + hyphens only)
    if [[ ! "$name" =~ ^[a-zA-Z0-9][a-zA-Z0-9-]*$ ]]; then
        echo "Error: instrument name must match ^[a-zA-Z0-9][a-zA-Z0-9-]*$" >&2
        exit 1
    fi

    # Check for duplicate
    if [[ -d "$env_dir" ]]; then
        echo "Error: instrument '$name' already exists at $env_dir" >&2
        exit 1
    fi

    # Check template exists (per Pitfall 4)
    if [[ ! -f "$template" ]]; then
        echo "Error: env.template not found at $template" >&2
        echo "Run install.sh first to deploy the template." >&2
        exit 1
    fi

    # Check default instance exists (for API key copy per D-04)
    local default_env="$CONFIG_DIR/default/env"
    if [[ ! -f "$default_env" ]]; then
        echo "Error: default instance env not found at $default_env" >&2
        echo "A default instance must exist to copy the API key from." >&2
        exit 1
    fi

    # Clone repo (per D-06)
    echo "Cloning $git_url to $work_dir..."
    git clone "$git_url" "$work_dir"

    # Create env from template (per D-03)
    mkdir -p "$env_dir"
    cp "$template" "$env_dir/env"

    # Replace placeholders (same pattern as install.sh)
    sed_inplace "s|INSTANCE_PLACEHOLDER|$name|g" "$env_dir/env"
    sed_inplace "s|WORKING_DIR_PLACEHOLDER|$work_dir|g" "$env_dir/env"
    sed_inplace "s|HOME_PLACEHOLDER|$HOME|g" "$env_dir/env"

    # Copy API key from default instance (per D-04)
    local api_key
    api_key=$(grep '^ANTHROPIC_API_KEY=' "$default_env" | cut -d= -f2)
    sed_inplace "s|^ANTHROPIC_API_KEY=.*|ANTHROPIC_API_KEY=$api_key|" "$env_dir/env"

    # Copy PATH line from default instance (per Pitfall 5 / Research Open Question 3)
    local default_path
    default_path=$(grep '^PATH=' "$default_env")
    if [[ -n "$default_path" ]]; then
        sed_inplace "s|^PATH=.*|$default_path|" "$env_dir/env"
    fi

    chmod 600 "$env_dir/env"

    # Enable systemd units (per D-03, D-10 / WDOG-05)
    systemctl --user daemon-reload
    systemctl --user enable --now "claude@${name}.service"
    systemctl --user enable --now "claude-watchdog@${name}.timer"

    echo "Instrument '$name' added and started"
    echo "  Service: claude@${name}.service"
    echo "  Watchdog: claude-watchdog@${name}.timer"
    echo "  Config: $env_dir/env"
    echo "  Working dir: $work_dir"
}
```

**5. Add `do_remove` function** (per D-07, D-08, D-10):
```bash
do_remove() {
    local name="$1"
    local env_dir="$CONFIG_DIR/$name"
    local work_dir="$HOME/instruments/$name"

    # Prevent removing default instance
    if [[ "$name" == "default" ]]; then
        echo "Error: cannot remove the default instance" >&2
        exit 1
    fi

    # Check instrument exists
    if [[ ! -d "$env_dir" ]]; then
        echo "Error: instrument '$name' not found" >&2
        exit 1
    fi

    # Stop and disable systemd units (per D-07)
    systemctl --user stop "claude@${name}.service" 2>/dev/null || true
    systemctl --user disable "claude@${name}.service" 2>/dev/null || true
    systemctl --user stop "claude-watchdog@${name}.timer" 2>/dev/null || true
    systemctl --user disable "claude-watchdog@${name}.timer" 2>/dev/null || true

    # Delete config directory (per D-07)
    rm -rf "$env_dir"

    # Delete working directory (per D-07)
    rm -rf "$work_dir"

    echo "Instrument '$name' removed"
    echo "  Stopped and disabled: claude@${name}.service, claude-watchdog@${name}.timer"
    echo "  Deleted: $env_dir"
    echo "  Deleted: $work_dir"
}
```
